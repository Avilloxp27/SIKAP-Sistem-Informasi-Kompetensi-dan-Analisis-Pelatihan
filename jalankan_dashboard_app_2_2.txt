@echo off
setlocal

REM ============================================================
REM Launcher Dashboard Evaluasi Kompetensi Pegawai v2.2
REM Lokasi project:
REM C:\Users\d04nr\OneDrive - Kemenkeu\Kantor\Diklat\PJJ Data Analitik\Action Learning\Sharing\prj_v2.2
REM ============================================================

title Dashboard Evaluasi Kompetensi Pegawai v2.2

set "PROJECT_DIR=C:\Users\d04nr\OneDrive - Kemenkeu\Kantor\Diklat\PJJ Data Analitik\Action Learning\Sharing\prj_v2.2"
set "APP_FILE=app_2.2.py"
set "REQ_FILE=requirements_2.2.txt"
set "VENV_DIR=.venv"

echo.
echo ============================================================
echo  Dashboard Evaluasi Kompetensi Pegawai v2.2
echo ============================================================
echo.

REM Pindah ke folder project.
cd /d "%PROJECT_DIR%"
if errorlevel 1 (
    echo [ERROR] Folder project tidak ditemukan:
    echo %PROJECT_DIR%
    echo.
    pause
    exit /b 1
)

REM Cek file aplikasi.
if not exist "%APP_FILE%" (
    echo [ERROR] File aplikasi tidak ditemukan: %APP_FILE%
    echo Pastikan file ini ada di folder project.
    echo.
    pause
    exit /b 1
)

REM Cek Python.
python --version >nul 2>&1
if errorlevel 1 (
    echo [ERROR] Python tidak ditemukan di PATH.
    echo Install Python dulu, lalu centang "Add Python to PATH".
    echo.
    pause
    exit /b 1
)

REM Buat virtual environment jika belum ada.
if not exist "%VENV_DIR%\Scripts\python.exe" (
    echo [INFO] Membuat virtual environment...
    python -m venv "%VENV_DIR%"
    if errorlevel 1 (
        echo [ERROR] Gagal membuat virtual environment.
        echo.
        pause
        exit /b 1
    )
)

REM Aktifkan virtual environment.
call "%VENV_DIR%\Scripts\activate.bat"
if errorlevel 1 (
    echo [ERROR] Gagal mengaktifkan virtual environment.
    echo.
    pause
    exit /b 1
)

REM Upgrade pip seperlunya. Tidak usah dramatis, pip.
echo [INFO] Menyiapkan pip...
python -m pip install --upgrade pip

REM Install dependency jika requirements tersedia.
if exist "%REQ_FILE%" (
    echo [INFO] Menginstall / mengecek dependency dari %REQ_FILE%...
    pip install -r "%REQ_FILE%"
    if errorlevel 1 (
        echo [ERROR] Gagal install dependency.
        echo Periksa koneksi internet atau package yang bermasalah.
        echo.
        pause
        exit /b 1
    )
) else (
    echo [WARNING] %REQ_FILE% tidak ditemukan. Melewati install dependency.
)

REM Jalankan aplikasi Streamlit.
echo.
echo [INFO] Menjalankan Streamlit...
echo Jika browser tidak terbuka otomatis, buka alamat yang muncul di terminal.
echo.
streamlit run "%APP_FILE%"

echo.
echo Aplikasi berhenti.
pause

endlocal
